Release Notes 
-------------

Version 0.13a         22/03/2010
--------------------------------

	* Se cambia la tabla "obras_sociales" por "coberturas_sociales", la cual tiene el campo "tipo_cobertura" 
		que	identifica si el registro corresponde a una obra social o a un plan social. Tambien se cambia el
		campo "id_obra_social" por "id_cobertura_social" de las tablas de las 4 tablas de afiliados. La
		estructura de la nueva tabla es:
		
			CREATE TABLE coberturas_sociales (
				id_cobertura_social 	SMALLINT 	UNSIGNED 	NOT NULL 	AUTO_INCREMENT,
				rnos		 			CHAR(6) 				NULL,
				nom_cobertura_social	VARCHAR(255)			NOT NULL,
				tipo_cobertura			CHAR(1)					NOT NULL 	COMMENT		'O=obra social; P=plan social',
				PRIMARY KEY(id_cobertura_social),
				INDEX idx_nom_cobertura_social(nom_cobertura_social),
				UNIQUE INDEX idx_unique_rnos(rnos)
			)ENGINE=MyISAM;
			
		NOTA: se actualizan los scripts "migra_padron2_nacionales.sql" y "generar_provin_afos.sql" aunque no se
			probaron ni todavia se ha incluido planes sociales a la nueva tabla
			
	* Se agrega al script "migra_padron2_nacionales.sql" el comando para actualizar siguiente registro
		que viene con un "enter" donde va una "ñ" y luego el proceso de generacion del "afos" para
		cobol falla. El comando es el siguiente:
		
		UPDATE afiliados_nacionales SET apellido='IÑIGUEZ' WHERE nro_doc='25992436';
		
	* Se crea el script "generar_provin_afos.sql" para generar el archivo "provin.txt" para el HMI
	
	* Se crea la tabla "afos" con la siguiente estructura:
	
		id 			BIGINT(11) 	NOT NULL auto_increment,
		codigo			CHAR(6) 	NOT NULL,	/* Código de Obra Social */
		tipodoc			CHAR(2)		NOT NULL,
		nrodoc			CHAR(8)		NOT NULL,
		apenom			CHAR(25)	NOT NULL,
		fechanac		CHAR(8) 	NOT NULL,
		condicion		CHAR(1)		NOT NULL,
		codipostal		CHAR(4) 	NOT NULL,
		codpro			CHAR(2)		NOT NULL,
		sexo			CHAR(1)		NOT NULL,
		peapor			CHAR(4) 	NOT NULL,
		cuiltit			CHAR(11)	NOT NULL,
	    PRIMARY KEY  (id),
		INDEX (nrodoc),
		INDEX (apenom),
		INDEX (codigo)

	* Se crean 4 tablas con identica estructura: 
	
		- afiliados
		
		- afiliados_nacionales

		donde "afiliados_xxxxxx" son tablas "MyISAM" y "afiliados" se define como la union de las
		otras 3 tablas con siguiente sentencia:
		
			CREATE TABLE afiliados (
				id_afiliado 		INTEGER(11) 					NOT NULL 	AUTO_INCREMENT,
				id_prov 			TINYINT 		UNSIGNED		NOT NULL,
				id_condicion 		TINYINT 		UNSIGNED 		NOT NULL,
				id_obra_social 		SMALLINT 		UNSIGNED 		NOT NULL,
				id_tipo_doc 		TINYINT 		UNSIGNED 		NOT NULL,
				sexo	 			ENUM			('M','F','D') 	NOT NULL,
				nro_doc 			INTEGER 		UNSIGNED 		NOT NULL,
				apellido 			VARCHAR(255) 					NOT NULL,
				nombre 				VARCHAR(255) 					NOT NULL,
				fecha_nac 			DATE 							NOT NULL,
				codigo_postal 		VARCHAR(8) 						NOT NULL,
				periodo_aporte 		CHAR(6)							NOT NULL,
				cuil_titular 		BIGINT 			UNSIGNED 		NULL,
				cuit_empresa 		BIGINT 			UNSIGNED 		NULL,
				PRIMARY KEY(id_afiliado),
				INDEX idx_apellido(apellido),
				INDEX idx_nombre(nombre),
				INDEX idx_id_obra_social(id_obra_social),
				INDEX idx_id_prov(id_prov),
				INDEX idx_nro_doc(nro_doc)
			)ENGINE=MERGE UNION=(afiliados_iapos,afiliados_nacionales,afiliados_profe);
